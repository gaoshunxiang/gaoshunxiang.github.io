    • 字典特征提取（特征离散化）
    #字典特征提取
# sklearn.feature_extraction.DictVectorizer(sparese=Ture,...)默认稀疏矩阵,使用稀疏矩阵节约内存
# 父类：转换器类
# DictVectorizer.fit_transform(X) X:array数据或者sparse矩阵 返回值为sparse矩阵
# DictVectorizer.inverse_transform(X) X:array数组或者sparse矩阵 返回值：转换之前的数据格式
# DictVectorizer.get_feature_name() 返回类别名称

#example

from sklearn.feature_extraction import DictVectorizer #导包
def dict_demo():
    data = [{'city':'北京','temperature':100},{'city':'上海','temperature':60},{'city':'深圳','temperature':30}]
    #1、实例化一个转换器
    transfer = DictVectorizer(sparse=False)
    
    #2、调用fit_transform()
    data_new = transfer.fit_transform(data)
    print("data_new:\n",data_new)
    print("特征名字:\n",transfer.get_feature_names())
    
    return None

    • 数值型数据的无量纲化：

      1. 归一化

      $$
      x^{‘}=\frac{x-min}{max-min} \quad and \quad x^{‘’}=x^{‘}*(max-min)+min
      $$

      sklearn.preprocessing.MinMaxScaler(feature_range=(0,1)...) #feature_range(放缩范围)# MinMaxScaler.fit_transform(x) x:numpy array格式的数据（n_samples,n_features）# 返回值：转化后的形状相同的array
      # example# 导入包from sklearn.preprocessing import MinMaxScalerdef minmax_demo():    #归一化      #1.获取数据    data = np.arange(15).reshape(5,3)    data = pd.DataFrame(data)    #2.实例化一个转换器 feature_range参数可设置归一化参数在什么范围    transfer = MinMaxScaler(feature_range=[2,3])    #3.调用fit_transfrom    data_new = transfer.fit_transform(data)    print("data_new\n",data_new)    return None# 归一化容易受到最大最小值影响，鲁棒性性较差
      • 注意：归一化容易受到最大最小值影响，鲁棒性性较差。
      1. 标准化
      sklearn.preprocessing.StandardScaler() # StandardScaler.fit_transform(x) x:numpy array格式的数据（n_samples,n_features）# 返回值：转化后的形状相同的array
      # example #标准化：（x-mean）/σdef stand_demo():    #1.获取数据    data = np.arange(15).reshape(5,3)    data = pd.DataFrame(data)    #2.实例化一个转换器 feature_range参数可设置归一化参数在什么范围    transfer = StandardScaler()    #3.调用fit_transfrom    data_new = transfer.fit_transform(data)    print("data_new\n",data_new)    return None
